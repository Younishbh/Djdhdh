const subjectColors = {
  'رياضيات': 's-math',
  'عربي': 's-arabic',
  'علوم': 's-science',
  'انجليزي': 's-english',
  'english': 's-english',

  'فن': 's-art',
};

function getSubjectClass(name) {
  if (!name) return 's-empty';
  const n = name.toLowerCase();
  for (const [key, cls] of Object.entries(subjectColors)) {
    if (n.includes(key.toLowerCase())) return cls;
  }
  return 's-studies';
}

function getGrade(pct) {
  if (pct >= 90) return {label:'ممتاز', cls:'grade-excellent'};
  if (pct >= 80) return {label:'جيد جداً', cls:'grade-very-good'};
  if (pct >= 70) return {label:'جيد', cls:'grade-good'};
  if (pct >= 60) return {label:'مقبول', cls:'grade-pass'};
  return {label:'راسب', cls:'grade-fail'};
}

function getGradeColor(pct) {
  if (pct >= 90) return '#1fa864';
  if (pct >= 80) return '#2e86ab';
  if (pct >= 70) return '#d4872a';
  if (pct >= 60) return '#e06c00';
  return '#e84855';
}

// DASHBOARD
// ============================================================
function renderDashboard() {
  const hw = appData.homeworks;
  const pending = hw.filter(h => h.status.includes('لم')).length;
  const done = hw.filter(h => h.status.includes('مسلم')).length;
  const exams = appData.exams;
  const avg = exams.length ? Math.round(exams.reduce((s,e) => s + (e.score/e.total*100),0) / exams.length) : 0;

  document.getElementById('stat-subjects').textContent = appData.subjects.length;
  document.getElementById('stat-hw-pending').textContent = pending;
  document.getElementById('stat-hw-done').textContent = done;
  document.getElementById('stat-avg').textContent = avg + '%';
  document.getElementById('badge-subjects').textContent = appData.subjects.length;
  document.getElementById('badge-hw').textContent = pending;
  document.getElementById('badge-exams').textContent = exams.length;

  // Today's schedule (Sunday=0, Mon=1, Tue=2, Wed=3, Thu=4)
  const dayMap = {0:'sun', 1:'mon', 2:'tue', 3:'wed', 4:'thu', 5:null, 6:null};
  const today = new Date().getDay();
  const todayKey = dayMap[today];
  const todayEl = document.getElementById('today-schedule');
  const dotColors = ['#2e86ab','#e84855','#1fa864','#6938c0','#d4872a','#0891b2','#c0357a'];

  if (!todayKey) {
    todayEl.innerHTML = '<div class="empty-state"><div class="icon">🌙</div><p>لا توجد حصص اليوم — استمتع بعطلتك!</p></div>';
  } else {
    const rows = appData.schedule.filter(r => r.period !== 'break' && r[todayKey]);
    todayEl.innerHTML = rows.map((r, i) => `
      <div class="today-item">
        <div class="today-time">${r.from} – ${r.to}</div>
        <div class="today-dot" style="background:${dotColors[i % dotColors.length]}"></div>
        <div class="today-subject">${r[todayKey]}</div>
        <span class="subject-badge ${getSubjectClass(r[todayKey])}" style="font-size:11px">ح${r.period}</span>
      </div>
    `).join('');
  }

  // Subject performance
  const perfMap = {};
  exams.forEach(e => {
    if (!perfMap[e.subject]) perfMap[e.subject] = {total:0, max:0, count:0};
    perfMap[e.subject].total += e.score;
    perfMap[e.subject].max += e.total;
    perfMap[e.subject].count++;
  });
  const barColors = ['#2e86ab','#1fa864','#d4872a','#e84855','#6938c0'];
  const perfEl = document.getElementById('subject-performance');
  perfEl.innerHTML = Object.entries(perfMap).map(([subj, data], i) => {
    const pct = Math.round(data.total / data.max * 100);
    const color = barColors[i % barColors.length];
    return `
      <div class="subject-bar">
        <div class="subject-bar-label">${subj}</div>
        <div class="bar-track">
          <div class="bar-fill" style="width:${pct}%;background:${color}"></div>
        </div>
        <div class="bar-pct" style="color:${color}">${pct}%</div>
      </div>
    `;
  }).join('') || '<div class="empty-state"><p>لا توجد بيانات اختبارات</p></div>';

  // Recent HW
  const recentHW = appData.homeworks.slice(-3).reverse();
  document.getElementById('recent-hw').innerHTML = recentHW.map(h => {
    let badge, dot;
    if (h.status.includes('مسلم')) { badge = 'success'; dot = '✅'; }
    else if (h.status.includes('قيد')) { badge = 'warning'; dot = '🔄'; }
    else { badge = 'danger'; dot = '❌'; }
    return `
      <div class="today-item">
        <div class="today-subject">${h.subject} — ${h.title}</div>
        <span class="badge ${badge}">${dot} ${h.status}</span>
      </div>
    `;
  }).join('') || '<div class="empty-state"><p>لا توجد واجبات</p></div>';

  // Recent Exams
  const recentExams = appData.exams.slice(-3).reverse();
  document.getElementById('recent-exams').innerHTML = recentExams.map(e => {
    const pct = Math.round(e.score / e.total * 100);
    const g = getGrade(pct);
    return `
      <div class="today-item">
        <div class="today-subject">${e.subject} — ${e.type}</div>
        <span class="grade-circle ${g.cls}" style="font-size:12px;width:auto;padding:4px 10px;border-radius:8px">${pct}%</span>
      </div>
    `;
  }).join('') || '<div class="empty-state"><p>لا توجد اختبارات</p></div>';
}

    </tr>
  `).join('') || '<tr><td colspan="7" style="text-align:center;padding:40px;color:var(--text-light)">لا توجد دروس</td></tr>';

  // Filters
  const subjects = [...new Set(appData.subjects.map(s => s.subject))];

}

function renderProgress() {
  const subjects = [...new Set(appData.exams.map(e => e.subject))];
  const colors = ['#2e86ab','#1fa864','#d4872a','#e84855','#6938c0','#0891b2'];

  // Progress rings
  const grid = document.getElementById('progress-grid');
  grid.innerHTML = subjects.map((subj, i) => {
    const subExams = appData.exams.filter(e => e.subject === subj);
    const totalScore = subExams.reduce((s,e) => s + e.score, 0);
    const totalMax = subExams.reduce((s,e) => s + e.total, 0);
    const pct = Math.round(totalScore / totalMax * 100);
    const color = colors[i % colors.length];
    const r = 36, circ = 2 * Math.PI * r;
    const offset = circ - (pct / 100) * circ;
    return `
      <div class="progress-card">
        <div class="progress-ring-wrap">
          <svg class="progress-ring" width="90" height="90" viewBox="0 0 90 90">
            <circle class="ring-bg" cx="45" cy="45" r="${r}"/>
            <circle class="ring-fill" cx="45" cy="45" r="${r}"
              stroke="${color}"
              stroke-dasharray="${circ}"
              stroke-dashoffset="${offset}"/>
          </svg>
          <div class="ring-label" style="color:${color}">${pct}%</div>
        </div>
        <div class="progress-subject">${subj}</div>
        <div class="progress-detail">${subExams.length} اختبار • ${totalScore}/${totalMax}</div>
      </div>
    `;
  }).join('');

  // Summary table
  const summaryEl = document.getElementById('summary-body');
  summaryEl.innerHTML = subjects.map(subj => {
    const subExams = appData.exams.filter(e => e.subject === subj);
    const totalScore = subExams.reduce((s,e) => s + e.score, 0);
    const totalMax = subExams.reduce((s,e) => s + e.total, 0);
    const pct = Math.round(totalScore / totalMax * 100);
    const g = getGrade(pct);
    const color = getGradeColor(pct);
    return `
      <tr>
        <td><span class="subject-badge ${getSubjectClass(subj)}">${subj}</span></td>
        <td style="text-align:center">${subExams.length}</td>
        <td style="text-align:center;font-weight:700;color:${color}">${totalScore}</td>
        <td style="text-align:center;color:var(--text-light)">${totalMax}</td>
        <td>
          <div class="pct-bar">
            <div class="pct-track"><div class="pct-fill" style="width:${pct}%;background:${color}"></div></div>
            <div class="pct-num" style="color:${color}">${pct}%</div>
          </div>
        </td>
        <td><span class="grade-circle ${g.cls}">${g.label}</span></td>
      </tr>
    `;
  }).join('');
}

function loadExcel(input) {
  const file = input.files[0];
  if (!file) return;
  document.getElementById('loadingOverlay').classList.add('show');

  const reader = new FileReader();
  reader.onload = function(e) {
    try {
      const data = new Uint8Array(e.target.result);
      const wb = XLSX.read(data, {type:'array'});

      // بيانات الطالب
      if (wb.SheetNames.includes('بيانات_الطالب')) {
        const ws = wb.Sheets['بيانات_الطالب'];
        const rows = XLSX.utils.sheet_to_json(ws, {header:1});
        rows.forEach(row => {
          if (row[0] && row[1]) {
            const label = String(row[0]);
            const val = String(row[1]);
            if (label.includes('اسم')) appData.student.name = val;
            if (label.includes('مرحلة')) appData.student.grade = val;
            if (label.includes('فصل الدراسي')) appData.student.term = val;
            if (label.includes('مدرسة')) appData.student.school = val;
          }
        });
        document.getElementById('sidebar-name').textContent = appData.student.name;
        document.getElementById('sidebar-grade').textContent = appData.student.grade + ' • الفصل ' + appData.student.term;
      }

      // جدول الحصص
      if (wb.SheetNames.includes('جدول_الحصص')) {
        const ws = wb.Sheets['جدول_الحصص'];
        const rows = XLSX.utils.sheet_to_json(ws, {header:1});
        const newSchedule = [];
        for (let i = 3; i < rows.length && i < 11; i++) {
          const r = rows[i];
          if (!r || r.length < 2) continue;
          const period = r[0];
          if (String(r[8] || '').includes('استراحة') || String(period) === '') {
            newSchedule.push({period:'break', sun:'', mon:'', tue:'', wed:'', thu:'', from:String(r[6]||''), to:String(r[7]||''), note:'استراحة'});
          } else {
            newSchedule.push({
              period: period,
              sun: String(r[1]||''), mon: String(r[2]||''), tue: String(r[3]||''),
              wed: String(r[4]||''), thu: String(r[5]||''),
              from: String(r[6]||''), to: String(r[7]||''), note: String(r[8]||'')
            });
          }
        }
        if (newSchedule.length > 0) appData.schedule = newSchedule;
      }

      // المواد والدروس
      if (wb.SheetNames.includes('المواد_والدروس')) {
        const ws = wb.Sheets['المواد_والدروس'];
        const rows = XLSX.utils.sheet_to_json(ws, {header:1, defval:''});
        const newSubjects = [];
        for (let i = 2; i < rows.length; i++) {
          const r = rows[i];
          if (!r[0]) continue;
          newSubjects.push({subject:String(r[0]), lesson:String(r[1]||''), unit:String(r[2]||''), grade:String(r[3]||''), teacher:String(r[4]||''), file:String(r[5]||''), note:String(r[6]||'')});
        }
        if (newSubjects.length > 0) appData.subjects = newSubjects;
      }

      // الواجبات
      if (wb.SheetNames.includes('الواجبات_والمهام')) {
        const ws = wb.Sheets['الواجبات_والمهام'];
        const rows = XLSX.utils.sheet_to_json(ws, {header:1, defval:''});
        const newHW = [];
        for (let i = 2; i < rows.length; i++) {
          const r = rows[i];
          if (!r[0] || String(r[0]).includes('اجمالي')) continue;
          newHW.push({subject:String(r[0]), title:String(r[1]||''), assigned:String(r[2]||''), due:String(r[3]||''), score:r[4]||0, status:String(r[5]||''), teacherNote:String(r[6]||''), studentNote:String(r[7]||'')});
        }
        if (newHW.length > 0) appData.homeworks = newHW;
      }

      // الاختبارات
      if (wb.SheetNames.includes('الاختبارات_والنتائج')) {
        const ws = wb.Sheets['الاختبارات_والنتائج'];
        const rows = XLSX.utils.sheet_to_json(ws, {header:1, defval:''});
        const newExams = [];
        for (let i = 2; i < rows.length; i++) {
          const r = rows[i];
          if (!r[0] || String(r[0]).includes('احصاء') || String(r[0]).includes('نسبة') || String(r[0]).includes('متوسط') || String(r[0]).includes('أعلى') || String(r[0]).includes('أدنى')) continue;
          if (typeof r[4] !== 'number') continue;
          newExams.push({subject:String(r[0]), type:String(r[1]||''), date:String(r[2]||''), total:Number(r[3])||20, score:Number(r[4])||0, note:String(r[8]||'')});
        }
        if (newExams.length > 0) appData.exams = newExams;
      }

      document.getElementById('loadingOverlay').classList.remove('show');
      showToast('✅', 'تم تحميل البيانات من Excel بنجاح!');
      renderSection(document.querySelector('.nav-item.active')?.getAttribute('onclick')?.match(/'(\w+)'/)?.[1] || 'dashboard');
    } catch(err) {
      document.getElementById('loadingOverlay').classList.remove('show');
      showToast('❌', 'خطأ في قراءة الملف: ' + err.message);
    }
  };
  reader.readAsArrayBuffer(file);
  input.value = '';
}

}
